# Uses python3
def edit_distance(s, t):
    # I am going to make an m x n array i.e. m rows and n cols
    m = len(s)
    n = len(t)
    if m == 0:
        return n
    elif n == 0:
        return m

    # Initializing an m x n list of lists to mimic the array
    E = [[j for j in range(n+1)] for i in range(m+1)]
    # Setting the base case
    for i in range(m+1):
        E[i][0] = i
    for i in range(1, m+1):
        for j in range(1, n+1):
            if s[i-1] == t[j-1]:
                E[i][j] = (min(E[i-1][j]+1,
                               E[i][j-1]+1,
                               E[i-1][j-1])
                           )
            else:
                E[i][j] = (min(E[i-1][j]+1,
                               E[i][j-1]+1,
                               E[i-1][j-1]+1)
                           )
    return E[m][n]

if __name__ == "__main__":
    print(edit_distance(input(), input()))
